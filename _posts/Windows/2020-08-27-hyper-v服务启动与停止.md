本文档介绍 hyper-v服务启动与停止 的相关内容。

```bash
@Echo Off
TITLE  Hyper-V 服务启动与停止

:start
CLS
COLOR 2f
:: 使用COLOR命令对控制台输出颜色进行更改
mode con cols=56 lines=20
:: MODE语句为设定窗体的宽和高
:sc_main
echo       -------------------------------------------
echo.&echo.
echo             1.启动Hyper-V服务 & echo.
echo             2.停止Hyper-V服务 & echo.
echo             q.退出 & echo.&echo.&echo.&echo.&echo.&echo.

set "select="
set/p select= 输入数字，按回车继续 :
if "%select%"=="1" (goto start_vmms)
if "%select%"=="2" (goto stop_vmms)
if "%select%"=="Q" (goto sc_exit)

:sc_exit
exit
goto :eof

:start_vmms
cls
::echo 显示当前处理的内容
net start vmms
::echo 显示处理后的结果
PAUSE >nul
Goto sc_main
:stop_vmms
cls
::echo 显示当前处理的内容
net stop vmms
::echo 显示处理后的结果
PAUSE >nul
Goto sc_main

```
