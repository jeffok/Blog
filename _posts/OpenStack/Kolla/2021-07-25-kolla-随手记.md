安装docker 需要先配置bridge 网卡，并指定ip地址 用于docker主机的ip地址
同时需要修改配置文件
使用镜像加速

sudo tee /etc/docker/daemon.json <<-'EOF'
{
"registry-mirrors": ["https://lshjxjt8.mirror.aliyuncs.com"],
"bridge": "br1"
}
EOF

[global]
index-url = http://pypi.douban.com/simple #豆瓣源，可以换成其他的源
trusted-host = pypi.douban.com #添加豆瓣源为可信主机，要不然可能报错
disable-pip-version-check = true #取消pip版本检查，排除每次都报最新的pip
timeout = 120

清华：https://pypi.tuna.tsinghua.edu.cn/simple

阿里云：http://mirrors.aliyun.com/pypi/simple/

华中理工大学：http://pypi.hustunique.com/

山东理工大学：http://pypi.sdutlinux.org/

[root@ceph01 ~]# cat /etc/docker/daemon.json
{"bip":"10.0.4.1/24"}

brctl addbr docker0
ip addr add 10.0.4.1/24 dev docker0
ip link set dev docker0 up

systemctl daemon-reload && systemctl enable docker && systemctl restart docker && systemctl status docker

pip install --ignore-installed <模块名称>
