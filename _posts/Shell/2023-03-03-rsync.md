# !/bin/bash

# repo: centos[base,updates,extras]

# repo: epel

nfjd_openstack_rsync(){
  Logfile=rsync.log
  Date=`date +"%F %T"`
  echo "${Date} nfjd_openstack_rsync" >> ${Logfile}
  sshpass='ssh -o StrictHostKeyChecking=no -p 8220'
  options='-auv --delete --bwlimit=5m --timeout=30 root@1.2.3.4'

  rsync --rsh="${sshpass}" ${options}:/root/ /data/filecoin/root &>> ${Logfile}

  echo '############################################' >> ${Logfile}
}

nfjd_openstack_rsync
